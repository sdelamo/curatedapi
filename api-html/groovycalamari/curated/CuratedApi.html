18 package groovycalamari.curated;
19 
20 import io.micronaut.http.annotation.Body;
21 import io.micronaut.http.annotation.PathVariable;
22 import io.micronaut.http.annotation.QueryValue;
23 import io.reactivex.Single;
24 import edu.umd.cs.findbugs.annotations.NonNull;
25 import javax.annotation.Nullable;
26 import javax.validation.Valid;
27 import javax.validation.constraints.Max;
28 import javax.validation.constraints.NotNull;
29 import javax.validation.constraints.Positive;
30 
31 /**
32  * @see <a href="http://support.curated.co/integrations/getting-started-with-the-curated-api/">Getting started with the Curated API</a>.
33  * @see <a href="http://support.curated.co/integrations/fetching-issue-data-with-the-api/">Fetching issue data with the API</a>
34  * @author Sergio del Amo
35  */
36 public interface CuratedApi {
37     String PATH_ISSUES = "/issues";
38     String PATH_EMAIL_SUBSCRIBERS = "/email_subscribers";
39 
40     Single<Issue> issue(@PathVariable @NonNull @NotNull @Positive Integer number);
41 
42     /**
43      * @param perPage How many issues to include in the results. The default value for this is 10 and the maximum value is 250.
44      * @param page Which page of data to retrieve. This will be affected by the per_page parameter above.
45      * @return List of issues
46      */
47     Single<Issues> issues(@QueryValue("per_page") @Nullable @Positive @Max(250) Integer perPage,
48                           @QueryValue @Nullable @Positive Integer page);
49 
50 
51     /**
52      * @param page Which page of data to retrieve. This will be affected by the per_page parameter above.
53      * @return List of issues
54      */
55     default Single<Issues> issues(@QueryValue @Nullable @Positive Integer page) {
56         return issues(null, page);
57 
58     }
59 
60     /**
61      * Fetches the list of issues
62      * @return List of issues
63      */
64     default Single<Issues> issues() {
65         return issues(null, null);
66     }
67 
68     Single<AddEmailSubscriberResponse> addEmailSubscriber(@Body @Valid @NotNull @NonNull AddEmailSubscriber subscriber);
69 }
